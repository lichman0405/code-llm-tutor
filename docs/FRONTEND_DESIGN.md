## 1. Technology Stack

### 1.1 Core Technology
- **Framework**: Next.js 14 (App Router)
- **Language**: TypeScript 5
- **Package Management**: pnpm 8

### 1.2 UI/UX
- **UI Component Library**: Shadcn/ui (based on Radix UI)
- **Styling Solution**: Tailwind CSS 3
- **Icons**: Lucide React
- **Font**: Inter (system sans-serif font)

### 1.3 Function Libraries
- **Code Editor**: Monaco Editor (VS Code kernel)
- **State Management**: Zustand 4
- **Form Handling**: React Hook Form + Zod
- **HTTP Client**: Axios
- **Charts**: Recharts
- **Animation**: Framer Motion (optional)
- **Markdown Rendering**: React Markdown

#### useAuth

```tsx
// hooks/useAuth.ts
import { create } from 'zustand';

interface AuthState {
  user: User | null;
  token: string | null;
  login: (email: string, password: string) => Promise<void>;
  logout: () => void;
  isAuthenticated: boolean;
}

export const useAuth = create<AuthState>((set) => ({
  user: null,
  token: null,
  isAuthenticated: false,
  
  login: async (email, password) => {
    const response = await fetch('/api/auth/login', {
      method: 'POST',
      body: JSON.stringify({ email, password }),
    });
    const { user, token } = await response.json();
    
    localStorage.setItem('token', token);
    set({ user, token, isAuthenticated: true });
  },
  
  logout: () => {
    localStorage.removeItem('token');
    set({ user: null, token: null, isAuthenticated: false });
  },
}));
```

#### useSubmission
```tsx
// hooks/useSubmission.ts
export function useSubmission() {
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [result, setResult] = useState<SubmissionResult | null>(null);

  const submit = async (problemId: string, code: string, language: string) => {
    setIsSubmitting(true);
    try {
      const response = await fetch('/api/submissions', {
        method: 'POST',
        body: JSON.stringify({ problem_id: problemId, code, language }),
      });
      const data = await response.json();
      setResult(data);
    } finally {
      setIsSubmitting(false);
    }
  };

  return { submit, isSubmitting, result };
}
```
